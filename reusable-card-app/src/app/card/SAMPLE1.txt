@Test
    void testTransferFile_DirectoryCreation() throws Exception {
        // Prepare data
        LocalDate date = LocalDate.now();
        List<CashAllocationModel> cashAllocationModels = new ArrayList<>();
        CashAllocationModel model = new CashAllocationModel();
        model.setInvoiceMatched("RSF-123");
        model.setCounterPartyId("CP123");
        model.setCreatedDate(new Date());

        // Initialize InvoicesAbList and InvoicesAb
        InvoicesAbList invoicesAbList = new InvoicesAbList();
        invoicesAbList.setInvoicesAb(new ArrayList<>());
        model.setInvoicesAbList(invoicesAbList);

        cashAllocationModels.add(model);

        // Access the private transferFile method using reflection
        Method transferFileMethod = AbDomain.class.getDeclaredMethod("transferFile", LocalDate.class, List.class);
        transferFileMethod.setAccessible(true);

        // Call the private method
        transferFileMethod.invoke(abDomain, date, cashAllocationModels);

        // Verify that the directory was created
        File dir = new File(".");
        assertTrue(dir.exists());
    }
