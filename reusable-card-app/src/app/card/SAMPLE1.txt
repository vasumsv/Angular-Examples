@Test
    void getPaymentLinesFromPaymentReference_CoversAllLines() {
        // Arrange
        String[] paymentReferences = {"PAYREF1"};

        List<SrgPaymentModel> srgPaymentModels = new ArrayList<>();
        SrgPaymentModel srgPaymentModel = new SrgPaymentModel();
        srgPaymentModel.setId(1L);
        srgPaymentModel.setProviderBdrId("PROVIDER1");
        srgPaymentModel.setPaymentCurrency("USD");
        srgPaymentModel.setSgEntity("ENTITY1");
        srgPaymentModel.setFeeCategory("FEE1");
        srgPaymentModel.setInvoiceReference("INV1,INV2");
        srgPaymentModel.setProviderName("PROVIDER_NAME");
        srgPaymentModel.setValueDate(LocalDate.now());
        srgPaymentModel.setSendingDate(LocalDateTime.now());
        srgPaymentModel.setPaymentAmount(100.0);
        srgPaymentModel.setPaymentType("TYPE1");
        srgPaymentModel.setDebitCredit("DEBIT");
        srgPaymentModel.setPaymentMode("MODE1");
        srgPaymentModel.setPaymentStatus("STATUS1");
        srgPaymentModel.setRcCode("RC1");
        srgPaymentModel.setComment("COMMENT1");
        srgPaymentModels.add(srgPaymentModel);

        List<String> ssiDetails = new ArrayList<>();
        ssiDetails.add("SSI1");
        ssiDetails.add("SSI2");

        when(srgMapper.findPaymentLinesFromPaymentReferences(paymentReferences)).thenReturn(srgPaymentModels);
        when(cashAllocationDomain.getSsiDetails("PROVIDER1", "USD", "ENTITY1", "FEE1")).thenReturn(ssiDetails);

        // Act
        List<PaymentLinesModel> result = cashAllocationDomain.getPaymentLinesFromPaymentReference(paymentReferences);

        // Assert
        assertNotNull(result);
        // We don't need to assert specific values, just that the method executes without exceptions
    }
