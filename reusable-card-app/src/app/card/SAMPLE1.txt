
    @Test
    void testUploadPaymentAbFileViaReferential_FileDeleted() throws IOException {
        // Create a temporary file
        File tempFile = File.createTempFile("test", ".txt");

        // Mock the uploadFileToGaiaServer method
        doNothing().when(referentialServiceApi).uploadFileToGaiaServer(any(), anyString(), anyString());

        // Call the method
        abDomain.uploadPaymentAbFileViaReferential(tempFile.getAbsolutePath());

        // Verify that the file was deleted
        assertFalse(tempFile.exists());

        // Clean up the temporary file (if the deletion failed)
        if (tempFile.exists()) {
            tempFile.delete();
        }
    }
